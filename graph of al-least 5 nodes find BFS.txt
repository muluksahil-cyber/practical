#include <stdio.h>
#include <conio.h>

#define MAX 20

int adj[MAX][MAX];
int visited[MAX];
int queue[MAX];
int front = -1, rear = -1;

// Function to add a vertex to the queue
void enqueue(int v)
{
    if (rear == MAX - 1)
    {
        printf("Queue Overflow!\n");
        return;
    }
    if (front == -1)
        front = 0;
    rear++;
    queue[rear] = v;
}

// Function to remove a vertex from the queue
int dequeue()
{
    int v;
    if (front == -1 || front > rear)
        return -1;
    v = queue[front];
    front++;
    return v;
}

// Breadth-First Search function
void BFS(int start, int vertices)
{
    int i, current;

    for (i = 0; i < vertices; i++)
        visited[i] = 0;

    enqueue(start);
    visited[start] = 1;

    printf("\nBFS Traversal starting from node %d: ", start);

    while ((current = dequeue()) != -1)
    {
        printf("%d ", current);

        for (i = 0; i < vertices; i++)
        {
            if (adj[current][i] == 1 && visited[i] == 0)
            {
                enqueue(i);
                visited[i] = 1;
            }
        }
    }
}

// Main function
void main()
{
    int vertices, edges;
    int i, j, src, dest, start;

    clrscr(); // clear the screen

    printf("=== Create Graph and Perform BFS Traversal ===\n");
    printf("Enter number of vertices (at least 5): ");
    scanf("%d", &vertices);

    if (vertices < 5)
    {
        printf("Graph must have at least 5 nodes!\n");
        getch();
        return;
    }

    printf("Enter number of edges: ");
    scanf("%d", &edges);

    // Initialize adjacency matrix
    for (i = 0; i < vertices; i++)
        for (j = 0; j < vertices; j++)
            adj[i][j] = 0;

    printf("\nEnter edges (source destination):\n");
    for (i = 0; i < edges; i++)
    {
        printf("Edge %d: ", i + 1);
        scanf("%d %d", &src, &dest);
        adj[src][dest] = 1;

        // Uncomment below for undirected graph
        // adj[dest][src] = 1;
    }

    printf("\nEnter starting node for BFS: ");
    scanf("%d", &start);

    BFS(start, vertices);

    getch(); // wait for key press before exit
}
